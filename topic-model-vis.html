from gensim import models
from gensim import corpora
import pyLDAvis.gensim

import os
import pickle
from tqdm.auto import tqdm

os.environ['NUMEXPR_MAX_THREADS'] = '31'
pyLDAvis.enable_notebook()

def build_lda_vis(dictionary_f, corpus_f, model_f):
    dictionary = corpora.Dictionary.load(dictionary_f)
    corpus = corpora.mmcorpus.MmCorpus(corpus_f)
    model = models.ldamodel.LdaModel.load(model_f)
    return pyLDAvis.gensim.prepare(model, corpus, dictionary)

%%time
years = pickle.load(open('data/lda/byyear/years.pkl', 'rb'))
logger = logging.getLogger('topic-model-vis')
by_year_prefix = "data/lda/byyear/"
for ts in tqdm(years):
    try:
        logger.info("Now working on {}".format(ts.year))
        dictionary_f = by_year_prefix + "{}.dict".format(ts.strftime("%Y-%m-%d"))
        corpus_f = by_year_prefix + '{}.mm'.format(ts.strftime("%Y-%m-%d"))
        model_f = by_year_prefix + '{}.model'.format(ts.year)
        logger.info("Starting to prepare visualisation for {}, {}, {}, {}".format(ts.year, dictionary_f, corpus_f, model_f))
        vis = build_lda_vis(dictionary_f, corpus_f, model_f)
        logger.info("Saving {} to JSON".format(ts.year))
        pyLDAvis.save_json(vis, by_year_prefix + "{}.json".format(ts.year))
        logger.info("Saving {} to HTML".format(ts.year))
        pyLDAvis.save_html(vis, by_year_prefix + "{}.html".format(ts.year))
    except Exception as e:
        logger.error("Error occurred while processing {}, error was {}".format(ts.year, e))
